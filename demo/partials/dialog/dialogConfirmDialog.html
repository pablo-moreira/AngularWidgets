ConfirmDialog is integrated with confirm behavior and used as a replacement for the javascript confirmation utility.
function DialogController(puiGrowl) { var vm = this; vm.confirmDialog = null; vm.yesAction = function() { puiGrowl.showInfoMessage('World destroyed!', 'Error...'); }; vm.noAction = function() { puiGrowl.showInfoMessage('World not destroyed!', 'Ops...'); }; vm.dismissAction = function() { puiGrowl.showInfoMessage('Dismiss!', 'Ops...'); }; }
