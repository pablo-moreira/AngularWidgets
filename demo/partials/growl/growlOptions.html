Options message example
function GrowlController($puiGrowl) { var vm = this; vm.showStickyMessage = function() { $puiGrowl.showInfoMessage('Message', "Message remains until close icon clicked", { stick: true }); }; vm.showLife500Message = function() { $puiGrowl.showInfoMessage('Message', "Message with options life 500 ms", { life: 500 } ); } vm.showLife6000Message = function() { $puiGrowl.showInfoMessage('Message', "Message with options life 6000 ms", { life: 6000 } ); } vm.clearMessages = function() { $puiGrowl.clear(); } };
